894 void servos_drive (const geometry_msgs::Twist::ConstPtr& msg)
895 {
896  /* this subscription works on the active_board */
897 
898  int i;
899  float delta;
900  float temp_x, temp_y, temp_r;
901  float dir_x, dir_y, dir_r;
902  float speed[4];
903 
904  /* msg is a pointer to a Twist message: msg->linear and msg->angular each of which have members .x .y .z */
905  /* the subscriber uses the maths from: http://robotsforroboticists.com/drive-kinematics/ */
906 
907  ROS_INFO("[%5.2f %5.2f %5.2f] [%5.2f %5.2f %5.2f]",
908  msg->linear.x, msg->linear.y, msg->linear.z, msg->angular.x, msg->angular.y, msg->angular.z);
909 
910  if (_active_drive.mode == MODE_UNDEFINED) {
911  ROS_ERROR("drive mode not set");
912  return;
913  }
914  if ((_active_drive.mode < MODE_UNDEFINED) || (_active_drive.mode >= MODE_INVALID)) {
915  ROS_ERROR("unrecognized drive mode set %d", _active_drive.mode);
916  return;
917  }
918 
919  dir_x = ((msg->linear.x < 0) ? -1 : 1);
920  dir_y = ((msg->linear.y < 0) ? -1 : 1);
921  dir_r = ((msg->angular.z < 0) ? -1 : 1);
922 
923  temp_x = _abs(msg->linear.x);
924  temp_y = _abs(msg->linear.y);
925  temp_r = _abs(msg->angular.z);
926 
927  // temp_x = _smoothing (temp_x);
928  // temp_y = _smoothing (temp_y);
929  // temp_r = _smoothing (temp_r) / 2;
930 
931 
932  switch (_active_drive.mode) {
933  case MODE_ACKERMAN:
934  /*
935  with ackerman drive, steering is handled by a separate servo
936  we drive assigned servos exclusively by the linear.x
937  */
938  speed[0] = temp_x * dir_x;
939  speed[0] = _convert_mps_to_proportional(speed[0]);
940 
941  ROS_INFO("ackerman drive mode speed=%6.4f", speed[0]);
942  break;
943  case MODE_DIFFERENTIAL:
944  /*
945  with differential drive, steering is handled by the relative speed of left and right servos
946  we drive assigned servos by mixing linear.x and angular.z
947  we compute the delta effect for left and right components
948  we use the sign of the angular velocity to determine which is the faster / slower
949  */
950 
951 
952  /* the delta is the angular velocity * half the drive track */
953  delta = (temp_r * (_active_drive.track / 2));
954 
955  if (dir_r > 0) { // turning right
956  speed[0] = (temp_x + delta) * dir_x;
957  speed[1] = (temp_x - delta) * dir_x;
958  } else { // turning left
959  speed[0] = (temp_x - delta) * dir_x;
960  speed[1] = (temp_x + delta) * dir_x;
961  }
962 
963  speed[0] = _convert_mps_to_proportional(speed[0]);
964  speed[1] = _convert_mps_to_proportional(speed[1]);
965 
966  ROS_INFO("differential drive mode speed left=%6.4f right=%6.4f", speed[0], speed[1]);
967  break;
968  case MODE_MECANUM:
969  /*
970  with mecanum drive, steering is handled by the relative speed of left and right servos
971  with mecanum drive, lateral motion is handled by the rotation of front and rear servos
972  we drive assigned servos by mixing linear.x and angular.z and linear.y
973  */
974 
975  /* the delta is the angular velocity * half the drive track */
976  delta = (temp_r * (_active_drive.track / 2));
977 
978  if (dir_r > 0) { // turning right
979  speed[0] = speed[2] = (temp_x + delta) * dir_x;
980  speed[1] = speed[3] = (temp_x - delta) * dir_x;
981  } else { // turning left
982  speed[0] = speed[2] = (temp_x - delta) * dir_x;
983  speed[1] = speed[3] = (temp_x + delta) * dir_x;
984  }
985 
986  speed[0] -= temp_y * dir_y;
987  speed[3] -= temp_y * dir_y;
988  speed[1] += temp_y * dir_y;
989  speed[2] += temp_y * dir_y;
990 
991  speed[0] = _convert_mps_to_proportional(speed[0]);
992  speed[1] = _convert_mps_to_proportional(speed[1]);
993  speed[2] = _convert_mps_to_proportional(speed[2]);
994  speed[3] = _convert_mps_to_proportional(speed[3]);
995 
996  ROS_INFO("mecanum drive mode speed leftfront=%6.4f rightfront=%6.4f leftrear=%6.4f rightreer=%6.4f", speed[0], speed[1], speed[2], speed[3]);
997  break;
998  default:
999  break;
1000  }
1001 
1002  /* find all drive servos and set their new speed */
1003  for (i=0; i<16; i++) {
1004  if (_servo_configs[_active_board-1][i].mode_pos == POSITION_LEFTFRONT)
1005  _set_pwm_interval_proportional (i+1, speed[0]);
1006  if (_servo_configs[_active_board-1][i].mode_pos == POSITION_RIGHTFRONT)
1007  _set_pwm_interval_proportional (i+1, speed[1]);
1008  if (_servo_configs[_active_board-1][i].mode_pos == POSITION_LEFTREAR)
1009  _set_pwm_interval_proportional (i+1, speed[2]);
1010  if (_servo_configs[_active_board-1][i].mode_pos == POSITION_RIGHTREAR)
1011  _set_pwm_interval_proportional (i+1, speed[3]);
1012  }
1013 }
